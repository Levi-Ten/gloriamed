  `created_at` TIMESTAMP NULL DEFAULT NULL,

CREATE TABLE hemograma (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    cnam_id BIGINT UNSIGNED NOT NULL,
    hemograma_checkbox TINYINT(1) DEFAULT 0,
    proba_hemograma_checkbox TINYINT(1) DEFAULT 0,
    rezultat_hemograma_file VARCHAR(255),
    rezultat_hemograma_text TEXT,
    vsh_checkbox TINYINT(1) DEFAULT 0,
    rezultat_vsh_file VARCHAR(255),
    cuagolograma_checkbox TINYINT(1) DEFAULT 0,
    rezultat_cuagolograma_file VARCHAR(255),
    FOREIGN KEY (cnam_id) REFERENCES cnam(id) ON DELETE CASCADE
);
