CREATE TABLE `cnam` (
  `id` BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  `numele` VARCHAR(255) NOT NULL,
  `prenumele` VARCHAR(255) NOT NULL,
  `data_nasterii` DATE NOT NULL,
  `idnp` VARCHAR(20) NOT NULL UNIQUE,
  `localitatea` VARCHAR(255) NOT NULL,
  `sectorul` VARCHAR(255) NULL,
  `strada` VARCHAR(255) NULL,
  `casa` VARCHAR(50) NULL,
  `blocul` VARCHAR(50) NULL,
  `apartamentul` VARCHAR(50) NULL,
  `full_info` VARCHAR(500) NOT NULL,
  `created_at` TIMESTAMP NULL DEFAULT NULL,
  `updated_at` TIMESTAMP NULL DEFAULT NULL
);
